import 'package:flutter/material.dart';
import 'package:todo_app_blue/AppColour.dart';
import 'package:todo_app_blue/Profile.dart';

import 'package:todo_app_blue/addTask.dart';

import 'HomePage.dart';

class SchoolManagement extends StatefulWidget {
  @override
  _SchoolManagementState createState() => _SchoolManagementState();
}

class _SchoolManagementState extends State<SchoolManagement> {
  int _selectedItemIndex = 0;
  final List pages = [
    HomePage(),
    addTask(),
    Profile(),
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
        bottomNavigationBar: BottomNavigationBar(
          elevation: 5,
          backgroundColor: Colors.white,
          unselectedItemColor: Colors.black26,
          selectedItemColor: Colors.black,
          selectedIconTheme: IconThemeData(color: AppColor.orange2),
          currentIndex: _selectedItemIndex,
          type: BottomNavigationBarType.fixed,
          onTap: (int index) {
            setState(() {
              _selectedItemIndex = index;
            });
          },
          items: [
            BottomNavigationBarItem(
              title: Text(""),
              icon: Icon(Icons.home),
            ),
            BottomNavigationBarItem(
              title: Text(""),
              icon: Icon(Icons.add),
            ),
            BottomNavigationBarItem(
              title: Text(""),
              icon: Icon(Icons.person),
            ),
          ],
        ),
        body: pages[_selectedItemIndex]);
  }
}
